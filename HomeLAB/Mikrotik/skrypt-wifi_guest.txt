/system script
add comment="Przelacznik sieci WiFi HomeNet_Guest" dont-require-permissions=\
    no name=toggle-guest-wifi owner=user policy=\
    ftp,reboot,read,write,policy,test,password,sniff,sensitive,romon source="\
    \n    :local guestInterfaces [/interface/wifi/find where name~\"guest\"]\
    \n    :if ([:len \$guestInterfaces] > 0) do={\
    \n        :foreach id in=\$guestInterfaces do={\
    \n            :local intName [/interface/wifi/get \$id name]\
    \n            :if ([/interface/wifi/get \$id disabled]) do={\
    \n                /interface/wifi/enable \$id\
    \n                :log warning \"Automatyzacja WiFi: Siec goscinna (\$intN\
    ame) zostala WLACZONA.\"\
    \n            } else={\
    \n                /interface/wifi/disable \$id\
    \n                :log warning \"Automatyzacja WiFi: Siec goscinna (\$intN\
    ame) zostala WYLACZONA.\"\
    \n            }\
    \n        }\
    \n    } else={\
    \n        :log error \"Automatyzacja WiFi: Krytyczny blad! Nie znaleziono \
    interfejsow zawierajacych slowo 'guest' w nazwie.\"\
    \n    }\
    \n"
